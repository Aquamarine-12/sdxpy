  • Construct an AST fragment that mirrors this
def make_count(name):
    return ast.Expr(
        value = ast.Call(
            func=ast.Name(id="count", ctx=ast.Load()),
            args=[ast.Constant(value=name)],
            keywords=[]
        )
    )

constructed = make_count("test")

Expr(

  • Modify source code
def modify(text):
    code = ast.parse(text)
    for node in ast.walk(code):
        if isinstance(node, ast.FunctionDef):
            node.body = [make_count(node.name), *node.body]
    return ast.unparse(code)

  • Create a counter
class CountCalls:
    def __init__(self):
        self.count = Counter()

    def __call__(self, name):
        self.count[name] += 1
  • Test it
